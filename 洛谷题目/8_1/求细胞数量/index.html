    жңҖеӨ§ \(100 \times 100 = 10000\) ж¬Ўж“ҚдҪңпјҢиҖ—ж—¶дёҚеҲ° 1 жҜ«з§’пјҢжһҒе…¶й«ҳж•ҲгҖӮ

    #include <iostream>
#include <vector>
#include <string>

using namespace std;

int n, m;
vector<string> grid;

// ж–№еҗ‘ж•°з»„пјҡдёҠгҖҒдёӢгҖҒе·ҰгҖҒеҸі
int dx[4] = {-1, 1, 0, 0};
int dy[4] = {0, 0, -1, 1};

// ж·ұеәҰдјҳе…ҲжҗңзҙўпјҢз”ЁдәҺвҖңжҠ№йҷӨвҖқиҝһйҖҡеқ—
void dfs(int x, int y) {
    // 1. е°ҶеҪ“еүҚи®ҝй—®зҡ„з»Ҷиғһж Үи®°дёә '0'пјҢйҒҝе…ҚйҮҚеӨҚи®ҝй—®
    grid[x][y] = '0';

    // 2. йҒҚеҺҶеӣӣдёӘж–№еҗ‘еҜ»жүҫзӣёйӮ»зҡ„з»ҶиғһйғЁеҲҶ
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];

        // 3. жЈҖжҹҘжҳҜеҗҰи¶Ҡз•ҢпјҢдё”ж–°дҪҚзҪ®жҳҜеҗҰд№ҹжҳҜз»Ҷиғһ ('1'~'9')
        if (nx >= 0 && nx < n && ny >= 0 && ny < m && grid[nx][ny] != '0') {
            dfs(nx, ny); // з»§з»ӯж·ұе…Ҙи”“е»¶
        }
    }
}

int main() {
    // дјҳеҢ–иҫ“е…Ҙиҫ“еҮәйҖҹеәҰ
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    grid.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> grid[i]; // дёҖж¬ЎиҜ»еҸ–дёҖж•ҙиЎҢеӯ—з¬ҰдёІ
    }

    int cell_count = 0;

    // йҒҚеҺҶж•ҙдёӘзҪ‘ж јеҜ»жүҫз»Ҷиғһзҡ„иө·зӮ№
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            // еҰӮжһңйҒҮеҲ°дёҚдёә '0' зҡ„еӯ—з¬ҰпјҢиҜҙжҳҺжүҫеҲ°дәҶж–°з»Ҷиғһ
            if (grid[i][j] != '0') {
                cell_count++; // з»Ҷиғһж•°йҮҸ +1
                dfs(i, j);    // еҗҜеҠЁ DFS е°ҶиҝҷдёӘз»ҶиғһжүҖеңЁзҡ„иҝһйҖҡеқ—е…ЁйғЁеҸҳдёә '0'
            }
        }
    }

    cout << cell_count << "\n";

    return 0;
}

    еҰӮжһңдҪ з”ЁдәҢз»ҙзҡ„ int ж•°з»„ cin >> matrix[i][j] жқҘиҜ»еҸ–пјҢC++ дјҡжҠҠиҝҷдёҖж•ҙиЎҢзӣҙжҺҘеҪ“жҲҗдёҖдёӘи¶…еӨ§ж•ҙж•°еҗғиҝӣеҺ»пјҢеҜјиҮҙй”ҷиҜҜгҖӮ жӯЈзЎ®еҒҡжі•пјҡеҝ…йЎ»дҪҝз”Ё string жҲ–иҖ… char ж•°з»„пјҢйҖҗиЎҢиҜ»еҸ–еӯ—з¬ҰдёІпјҢжҲ–иҖ…йҖҗдёӘеӯ—з¬ҰиҜ»еҸ–гҖӮ

    вқҢ 2. еҝҳи®°е°Ҷж јеӯҗйҮҚзҪ®жҲ–ж Үи®°

    еңЁ dfs ејҖе§Ӣж—¶пјҢеҝ…йЎ»з«ӢеҲ»жү§иЎҢ grid[x][y] = '0';гҖӮ еҰӮжһңеҸӘжҳҜйҒҚеҺҶиҖҢдёҚдҝ®ж”№е®ғзҡ„еҖјпјҢеңЁ A иө°еҲ° B еҗҺпјҢB жүҫеӣӣе‘Ёж—¶еҸҲдјҡиө°еӣһ AпјҢд»ҺиҖҢеј•еҸ‘жӯ»еҫӘзҺҜпјҢжңҖз»ҲеҜјиҮҙеҶ…еӯҳзҲҶж»ЎпјҲStack Overflow / REпјүгҖӮ

    вқҢ 3. ж–ңеҗ‘дёҚиҝһйҖҡ

    йўҳзӣ®дёҘж ји§„е®ҡпјҡвҖңжІҝз»Ҷиғһж•°еӯ—дёҠдёӢе·ҰеҸіиӢҘиҝҳжҳҜз»Ҷиғһж•°еӯ—еҲҷдёәеҗҢдёҖз»ҶиғһвҖқгҖӮ д№ҹе°ұжҳҜиҜҙеҜ№и§’зәҝж–№еҗ‘пјҲе·ҰдёҠгҖҒе·ҰдёӢгҖҒеҸідёҠгҖҒеҸідёӢпјүжҳҜдёҚдә’йҖҡзҡ„гҖӮеҰӮжһңд№ жғҜдәҶе…«ж–№еҗ‘зҡ„жҗңзҙўйўҳпјҢж–№еҗ‘ж•°з»„еҲ«дёҚе°ҸеҝғејҖжҲҗдәҶ 8 дёӘгҖӮ

    рҹҡҖ дёҖеҸҘиҜқжҖ»з»“

    йҒҮеҲ°йқһ 0 е°ұи®Ўж•°еҠ дёҖпјҢз„¶еҗҺдёҖеҸ‘ DFS жҙӘж°ҙеЎ«е……жҠҠиҝһйҖҡзҡ„ж јеӯҗе…ЁжҠ№е№ідёә 0пјҢз»§з»ӯжүҫдёӢдёҖдёӘгҖӮ

    рҹ”Ҙ иЎҘе……пјҡзЎ¬ж ёзҹҘиҜҶзӮ№

    еҗҢдёҖйҒ“йўҳзҡ„дёҚеҗҢи§Ҷи§’пјҡDFS vs BFS vs е№¶жҹҘйӣҶ

    иҝҷйҒ“йўҳе…¶е®һжҳҜеӣҫи®әдёӯйқһеёёж ёеҝғзҡ„вҖңжұӮиҝһйҖҡеҲҶйҮҸвҖқй—®йўҳгҖӮйҷӨдәҶжҲ‘д»¬д»Јз ҒдёӯеҶҷзҡ„ DFSпјҢе®ғиҝҳжңүдёӨз§Қз»Ҹе…ёзҡ„йҷҚз»ҙжү“еҮ»и§Ҷи§’пјҡ

    1. BFS (е№ҝеәҰдјҳе…Ҳжҗңзҙў) и§Ҷи§’
    2. еҢәеҲ«пјҡDFS жҳҜдёҖжқЎи·Ҝиө°еҲ°й»‘пјҢBFS еҲҷжҳҜз”Ёж°ҙжіўжү©ж•Јзҡ„ж–№ејҸдёҖеұӮеұӮеҫҖеӨ–жүҫгҖӮ

    3. дјҳеҠЈпјҡиҝҷйҒ“йўҳ \(N \le 100\)пјҢDFS зҡ„йҖ’еҪ’ж·ұеәҰжңҖеӨ§дёҖдёҮеұӮпјҢеӢүејәдёҚдјҡзҲҶж ҲгҖӮдҪҶеҰӮжһңж•°жҚ®иҢғеӣҙжӢүеӨ§еҲ° \(N \le 2000\)пјҢDFS дјҡеӣ дёәи°ғз”Ёж ҲеӨӘж·ұзӣҙжҺҘеҜјиҮҙзЁӢеәҸеҙ©жәғпјҲREпјүгҖӮиҖҢ BFS дҫқиө–зҡ„жҳҜ std::queueпјҢеҶ…еӯҳеңЁе ҶеҢә (Heap) еҲҶй…ҚпјҢз»қдёҚдјҡзҲҶж ҲгҖӮеңЁзңҹжӯЈзҡ„е·Ҙдёҡзә§д»Јз ҒдёӯпјҢFlood Fill йҖҡеёёдјҳе…ҲдҪҝз”Ё BFSгҖӮ

    4. е№¶жҹҘйӣҶ (Disjoint Set Union, DSU) и§Ҷи§’

    5. жҖқи·ҜпјҡдёҖејҖе§ӢпјҢеҒҮи®ҫзҪ‘ж јйҮҢжүҖжңүзҡ„йқһ 0 ж•°еӯ—йғҪжҳҜдә’зӣёзӢ¬з«Ӣзҡ„вҖңеҚ•иә«жұүвҖқгҖӮжҲ‘д»¬йҒҚеҺҶзҪ‘ж јпјҢжҜҸж¬ЎжЈҖжҹҘеҪ“еүҚж јеӯҗзҡ„вҖңеҸіиҫ№вҖқе’ҢвҖңдёӢиҫ№вҖқгҖӮеҰӮжһңйӮ»еұ…д№ҹжҳҜйқһ 0пјҢжҲ‘д»¬е°ұжҠҠиҝҷдёӨдёӘж јеӯҗвҖңеҗҲе№¶вҖқеҲ°еҗҢдёҖдёӘйӣҶеҗҲйҮҢгҖӮ
    6. дјҳеҠЈпјҡе…ЁйғЁеҗҲе№¶е®ҢеҗҺпјҢеҸӘиҰҒж•°дёҖж•°е№¶жҹҘйӣҶйҮҢжңүеҮ дёӘвҖңзҘ–е®—иҠӮзӮ№вҖқпјҢе°ұиғҪеҫ—еҮәжңүеҮ дёӘзӢ¬з«Ӣзҡ„з»ҶиғһгҖӮиҝҷз§ҚеҶҷжі•е®Ңе…ЁйҒҝејҖдәҶжҗңзҙўдёҺйҖ’еҪ’пјҢеёёз”ЁдәҺеҠЁжҖҒиҝһйҖҡжҖ§й—®йўҳпјҲжҜ”еҰӮпјҡеҰӮжһңз»ҶиғһжҳҜеҠЁжҖҒдёҖдёӘдёӘй•ҝеҮәжқҘзҡ„пјҢй—®жҹҗдёӘж—¶еҲ»жңүеҮ дёӘиҝһйҖҡеқ—пјҢDFS/BFS жҜҸж¬ЎйғҪиҰҒйҮҚж–°жҗңпјҢиҖҢе№¶жҹҘйӣҶеҸҜд»Ҙ \(O(1)\) зһ¬й—ҙз»ҷеҮәзӯ”жЎҲпјүгҖӮ

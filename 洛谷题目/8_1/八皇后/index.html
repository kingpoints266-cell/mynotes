    if (row > n) {
        total_solutions++; // жүҫеҲ°дёҖдёӘи§Ј
        // йўҳзӣ®иҰҒжұӮеҸӘиҫ“еҮәеүҚ 3 дёӘи§Ј
        if (total_solutions <= 3) {
            for (int i = 1; i <= n; i++) {
                cout << path[i] << " ";
            }
            cout << "\n";
        }
        return;
    }

    // е°қиҜ•еңЁеҪ“еүҚиЎҢ row зҡ„з¬¬ i еҲ—ж”ҫзҪ®зҡҮеҗҺ
    for (int i = 1; i <= n; i++) {
        // жЈҖжҹҘиҜҘеҲ—гҖҒдёӨжқЎеҜ№и§’зәҝжҳҜеҗҰе®үе…Ё
        if (!col[i] && !diag1[row + i] && !diag2[row - i + n]) {
            // и®°еҪ•и·Ҝеҫ„
            path[row] = i;

            // еҚ дҪҚж Үи®°пјҲдҝқжҠӨзҺ°еңәпјү
            col[i] = true;
            diag1[row + i] = true;
            diag2[row - i + n] = true;

            // йҖ’еҪ’иҝӣе…ҘдёӢдёҖиЎҢ
            dfs(row + 1);

            // еӣһжәҜпјҲжҒўеӨҚзҺ°еңәпјүпјҡжҠҠеҲҡеҲҡж”ҫзҡ„зҡҮеҗҺжӢҝиө°пјҢе°қиҜ•дёӢдёҖеҲ—
            col[i] = false;
            diag1[row + i] = false;
            diag2[row - i + n] = false;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;

    // д»Һз¬¬ 1 иЎҢејҖе§Ӣжҗңзҙў
    dfs(1);

    // жңҖеҗҺиҫ“еҮәи§Јзҡ„жҖ»ж•°
    cout << total_solutions << "\n";

    return 0;
}

    вҡ пёҸ жҳ“й”ҷзӮ№

    вқҢ 1. еҜ№и§’зәҝж•°з»„ејҖеҫ—еӨӘе°ҸеҜјиҮҙи¶Ҡз•Ң

    иЎҢе’ҢеҲ—зҡ„жңҖеӨ§еҖјжҳҜ 13пјҢдҪҶеҜ№и§’зәҝзҡ„е…¬ејҸжҳҜ row + col е’Ң row - col + nгҖӮ е®ғд»¬зҡ„жңҖеӨ§еҖјдјҡиҫҫеҲ° \(2 \times N\)пјҢд№ҹе°ұжҳҜе°Ҷиҝ‘ 30гҖӮ еҰӮжһңеҜ№и§’зәҝзҡ„еҲӨе®ҡж•°з»„ diag1 е’Ң diag2 еҸӘејҖ 20 зҡ„еӨ§е°ҸпјҢдјҡеҜјиҮҙж•°з»„и¶Ҡз•Ңеҙ©жәғпјҲREпјүгҖӮе»әи®®ејҖеҲ° 40 зЎ®дҝқз»қеҜ№е®үе…ЁгҖӮ

    вқҢ 2. еҝҳи®°жҒўеӨҚзҺ°еңәпјҲеӣһжәҜпјү

    иҝҷжҳҜеӣһжәҜжі•жңҖиҮҙе‘Ҫзҡ„й”ҷиҜҜпјҡ

    col[i] = false; // еҝ…йЎ»еҶҷпјҒ

    еҰӮжһңдҪ иҝӣе…ҘйҖ’еҪ’еҗҺпјҢйҖҖеҮәжқҘж—¶дёҚжҠҠж Үи®°йҮҚж–°еҸҳеӣһ falseпјҢеҗҺйқўзҡ„жҗңзҙўе°ұдјҡд»ҘдёәиҝҷеҲ—иҝҳиў«еҚ зқҖпјҢеҜјиҮҙд»Җд№Ҳд№ҹжҗңдёҚеҲ°пјҢжңҖеҗҺиҫ“еҮәе…Ёдёә 0гҖӮ

    вқҢ 3. дё»еҜ№и§’зәҝзҡ„еҒҸз§»йҮҸ

    row - col жҳҜжңүеҸҜиғҪдә§з”ҹиҙҹж•°зҡ„пјҢжҜ”еҰӮеңЁз¬¬ 1 иЎҢз¬¬ 13 еҲ—пјҢ1 - 13 = -12гҖӮ C++ ж•°з»„дёӢж ҮдёҚиғҪдёәиҙҹж•°пјҢеҝ…йЎ»з»ҹдёҖеҠ дёҠдёҖдёӘеёёж•°пјҲиҝҷйҮҢз”ЁдәҶ nпјүпјҢеҶҷжҲҗ row - col + n жҠҠе®ғе№із§»еҲ°жӯЈж•°еҢәй—ҙгҖӮ

    рҹҡҖ дёҖеҸҘиҜқжҖ»з»“

    NзҡҮеҗҺ = йҖҗиЎҢ DFS + е·§еҰҷеҲ©з”ЁжЁӘзәөеқҗж ҮеҠ еҮҸи§„еҫӢзҡ„ O(1) еҶІзӘҒеҲӨе®ҡ + дёҘи°Ёзҡ„еӣһжәҜиҝҳеҺҹзҺ°еңәгҖӮ

    зЎ¬ж ёзҹҘиҜҶзӮ№пјҡNзҡҮеҗҺзҡ„жһҒйҷҗдјҳеҢ– вҖ”вҖ” дҪҚиҝҗз®—пјҲBitmaskпјү

    иҝҷйҒ“йўҳзҡ„ж•°жҚ®иҢғеӣҙжҳҜ \(n \le 13\)пјҢжҷ®йҖҡзҡ„еёғе°”ж•°з»„е®Ңе…ЁеӨҹз”ЁгҖӮдҪҶеҰӮжһңеңЁиҝӣйҳ¶жҜ”иөӣдёӯпјҢж•°жҚ®иҢғеӣҙжӢүй«ҳеҲ°дәҶ \(n \le 15\) з”ҡиҮі \(n \le 16\) ж—¶пјҢеёғе°”ж•°з»„зҡ„иҜ»еҶҷе’Ң for еҫӘзҺҜе°ұдјҡжҳҫеҫ—еӨӘж…ўиҖҢеҜјиҮҙ TLEпјҲи¶…ж—¶пјүгҖӮ

    иҝҷж—¶еҖҷпјҢз«һиөӣеӨ§дҪ¬д»¬дјҡзҘӯеҮәдҪҚиҝҗз®—дјҳеҢ–пјҡ з”ЁдёүдёӘж•ҙеһӢеҸҳйҮҸпјҲintпјүзҡ„дәҢиҝӣеҲ¶дҪҚжқҘд»Јжӣҝеёғе°”ж•°з»„гҖӮ * columnпјҡи®°еҪ•е“ӘдәӣеҲ—жңүзҡҮеҗҺпјҲдәҢиҝӣеҲ¶дҪҚдёә 1 иЎЁзӨәжңүпјүгҖӮ * diag1пјҡи®°еҪ•е·ҰдёӢеҜ№и§’зәҝзҡ„еҚ з”Ёжғ…еҶөгҖӮ * diag2пјҡи®°еҪ•еҸідёӢеҜ№и§’зәҝзҡ„еҚ з”Ёжғ…еҶөгҖӮ

    жңҖжғҠиүізҡ„дҪҚиҝҗз®—йӯ”жі•пјҡ еңЁеҜ»жүҫеҪ“еүҚиЎҢжңүе“ӘдәӣеҲ—еҸҜд»Ҙж”ҫзҡҮеҗҺж—¶пјҢеҸӘйңҖдёҖиЎҢд»Јз Ғпјҡ 

    int available_positions = ((1 << n) - 1) & ~(column | diag1 | diag2);
